const apps = db.getSiblingDB("db_mapp").apps.find({})
    .projection({})
    .sort({ _id: -1 })
    .limit(100).toArray()
// console.log(apps)
const insertedUsers = []
apps.forEach((app => {
    app.config.policy.groups.map(group => {
        console.log(group.roles)
        db.getSiblingDB("db_api").users.find({ "policy_role_master": { $in: group.roles } }).forEach(user => { insertedUsers.push({
                "updated_at": ISODate(),
                "created_at": ISODate(),
                "by_role": user.policy_role_master,
                "by_user": ObjectId("000000000000000000000000"),
                "user_id": user._id,
                "app_id": app._id
            }) })
    })
}))

console.log(insertedUsers)

db.getSiblingDB("db_mapp").subscribers.insertMany(insertedUsers)
