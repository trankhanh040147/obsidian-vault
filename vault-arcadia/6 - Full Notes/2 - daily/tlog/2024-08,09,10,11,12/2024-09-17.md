files := make([]models.TicketFile, 0)  
for _, attachment := range params.Attachments {  
    parsedURL, err := url.ParseQuery(attachment)  
    if err != nil {  
       logger.Error().Err(err).Int("statusCode", fiber.StatusInternalServerError).Str("function", "SubmitTicketDialog").  
          Str("functionInline", "url.ParseQuery(attachment)").Msg("requester-controller-command")  
       ctrl.service.sendBotServerProblem(msg.Sender, msg.SenderDevice.Platform)  
       return  
    }  
    fileId, _ := primitive.ObjectIDFromHex(parsedURL.Get("i"))  
    files = append(files, models.TicketFile{  
       Id:   fileId,  
       Name: parsedURL.Get("f"),  
    })  
}
